ALTER TABLE COMUNAS
ADD FOREIGN KEY (COMUNA_FK) REFERENCES colegios(ID);




SELECT NOMBRE 
FROM (
SELECT
    ID
    ,NOMBRE
    ,COUNT(1) OVER(PARTITION BY nombre_sede) TOTAL_COLEGIOS
FROM
    colegios
ORDER BY
    nombre_sede
;
)W
INNER JOIN COMUNAS ON W.nombre_sede = colegios.NOMBRE;
